# Getting the first 10 characters out of 'monthorweek'

data <- data %>%
  mutate(monthorweek = str_sub(monthorweek, 1, 10),
         monthorweek = ymd(monthorweek),
         monthorweek = floor_date(monthorweek, "month"))
# Standardizing the index variable by school name and keyword 

# Group here first 
grouped_schools <- data %>%
  group_by(schname, keyword) %>% 
  mutate(index_mean = mean(index),
         stdev_index = sd(index))

# Group standardization 
standardized_df <- grouped_schools %>%
  mutate(st_index = (index - index_mean) / stdev_index)

# This is now interpreted as: a one-unit change in standardized index is associated as a one-standard deviation change in search interest.

Aggregate the standardized index to the school-month level

agg_data <- standardized_df %>%
  group_by(schname, monthorweek) %>%
  summarize(agg_index = mean(st_index))
`summarise()` has grouped output by 'schname'. You can override using the
`.groups` argument.
# Addressing NA's
agg_data <- agg_data %>%
  drop_na(agg_index)
